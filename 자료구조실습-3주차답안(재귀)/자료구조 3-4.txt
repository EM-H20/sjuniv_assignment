#include <stdio.h>
#include <stdlib.h>
#define Max(x, y) ((x) > (y)) ? (x) : (y)

int MAX(int* arr, int n) {
	if (n == 1) 
		return arr[0];	
	else 
		return Max(MAX(arr, n - 1), arr[n - 1]);
}

int main() {
	int N;
	scanf("%d", &N);
	
	int* arr = (int*)malloc(sizeof(int) * N);
	for (int i = 0; i < N; i++)
		scanf("%d", &arr[i]);
	printf("%d", MAX(arr, N));

	free(arr);
}
