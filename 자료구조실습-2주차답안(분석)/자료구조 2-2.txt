#include <stdio.h>
#include <stdlib.h>

int mostOnes(int **arr, int n) {
	int j = 0, idx = 0, i = 0;
	
	while (j < n && i < n) {
		if (arr[i][j] == 1) {
			j++;
			idx = i;
		}
		else 
			i++;		
	}
	return idx;	
}
int main() {
	int** arr, N;
	scanf("%d", &N);

	arr = (int**)malloc(sizeof(int*) * N);
	for (int i = 0; i < N; i++)
		arr[i] = (int*)malloc(sizeof(int) * N);

	for (int i = 0; i < N; i++)
		for (int j = 0; j < N; j++)
			scanf("%d", &arr[i][j]);

	printf("%d\n", mostOnes(arr, N));

	for (int i = 0; i < N; i++)
		free(arr[i]);
	free(arr);
}
