#include <stdio.h>
#include <stdlib.h>
int* prefixAverages1(int* arr, int n) {//O(n^2)
	int* A = (int*)malloc(sizeof(int) * n);
	for (int i = 0; i < n; i++) {
		int sum = 0;
		for (int j = 0; j <= i; j++) {
			sum += arr[j];
		}
		A[i] = ((double)(sum) / (i + 1) + 0.5);
	}
	return A;
}
int* prefixAverages2(int* arr, int n) { //O(n)
	int* A = (int*)malloc(sizeof(int) * n);
	int sum = 0;
	for (int i = 0; i < n; i++) {
		sum += arr[i];
		A[i] = ((double)(sum) / (i + 1) + 0.5);
	}
	return A;
}
int main() {
	int* arr, N;
	scanf("%d", &N);

	arr = (int*)malloc(sizeof(int) * N);
	for (int i = 0; i < N; i++)
		scanf("%d", &arr[i]);

	int* A = prefixAverages1(arr, N);
	for (int i = 0; i < N; i++)
		printf("%d ", A[i]);

	printf("\n");
	A = prefixAverages2(arr, N);
	for (int i = 0; i < N; i++)
		printf("%d ", A[i]);

	free(A);
	free(arr);
}
