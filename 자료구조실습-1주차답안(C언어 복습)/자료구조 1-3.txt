#include <stdio.h>
#include <stdlib.h>

void swap(int* a, int* b) {
	int tmp = *a;
	*a = *b;
	*b = tmp;
}

int main() {
	int n;
	int* arr;
	int a, b;

	scanf("%d", &n);

	arr = (int*)malloc(sizeof(int) * n);
	if (arr == NULL)
		return -1;

	for (int i = 0; i < n; i++)
		scanf("%d", &arr[i]);

	scanf("%d %d", &a, &b);

	swap(&arr[a], &arr[b]);

	for (int i = 0; i < n; i++)
		printf("%2d", arr[i]);

	free(arr);
	return 0;
}
