#include <stdio.h>

int ABC(int* arr, int k) {
	int max = arr[k], idx = k;

	for (int i = k; i < 10; i++) {
		if (arr[i] > max) {
			max = arr[i];
			idx = i;
		}
	}
	int tmp = arr[k];
	arr[k] = arr[idx];
	arr[idx] = tmp;
}
int main() {
	int arr[10];

	for (int i = 0; i < 10; i++) {
		scanf("%d", &arr[i]);
	}

	for (int i = 0; i < 10; i++) {
		ABC(arr, i);
	}

	for (int i = 0; i < 10; i++) {
		printf("%2d", arr[i]);
	}
	return 0;
}
