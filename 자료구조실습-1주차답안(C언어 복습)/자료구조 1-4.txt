#include <stdio.h>
#include <stdlib.h>
#include <string.h>
int main() {
	char str[101]; //최대100, 널문자 + 1

	scanf("%s", str); //문자입력
	
	int str_len = (int)strlen(str);

	
	for (int i = 0; i < str_len; i++) {
		printf("%s\n", str);
		char tmp = str[0];

		for (int j = 0; j < str_len - 1; j++) 
			str[j] = str[j + 1];		

		str[str_len - 1] = tmp;		
	}	
	return 0;
}
